    <script>

        // DATA SOALAN KUIZ

        const quizData = {

            1: [ // TAHAP 1: Mengenal pasti mata wang (Syiling & Wang Kertas)

                { q: "Apakah warna bagi duit syiling 50 sen Malaysia?", a: ["Emas", "Perak", "Gangsa", "Biru"], c: 0 },

                { q: "Apakah gambar utama yang ada pada setiap wang kertas Malaysia?", a: ["KLCC", "Bunga Raya", "Harimau", "Yana"], c: 1 },

                { q: "Wang kertas RM10 mempunyai warna dominan...", a: ["Merah", "Hijau", "Kuning", "Ungu"], c: 0 }

            ],

            2: [ // TAHAP 2: Mewakilkan nilai wang (Sen hingga RM1)

                { q: "Dua keping duit syiling 20 sen dan satu keping 10 sen menjadi...", a: ["30 sen", "40 sen", "50 sen", "60 sen"], c: 2 },

                { q: "Berapakah keping syiling 10 sen yang diperlukan untuk menjadi RM1?", a: ["5 keping", "10 keping", "20 keping", "1 keping"], c: 1 },

                { q: "Adham ada 3 keping 20 sen. Berapakah jumlah wang Adham?", a: ["50 sen", "60 sen", "70 sen", "80 sen"], c: 1 },

                { q: "Dua keping syiling 50 sen adalah sama nilai dengan...", a: ["RM1", "50 sen", "RM2", "10 sen"], c: 0 },

                { q: "Antara kombinasi berikut, yang manakah bernilai 85 sen?", a: ["50 sen + 20 sen + 10 sen", "50 sen + 20 sen + 10 sen + 5 sen", "50 sen + 50 sen", "20 sen + 20 sen + 20 sen"], c: 1 }

            ],

            3: [ // TAHAP 3: Mewakilkan nilai wang (Ringgit hingga RM10)

                { q: "Sebutkan jumlah bagi sekeping RM5 dan dua keping RM1.", a: ["RM6", "RM7", "RM8", "RM10"], c: 1 },

                { q: "Berapakah keping wang RM2 yang sama nilai dengan sekeping RM10?", a: ["2 keping", "3 keping", "4 keping", "5 keping"], c: 3 },

                { q: "Kombinasi wang: RM5 + RM1 + RM1 + RM1. Berapakah jumlah nilainya?", a: ["RM6", "RM7", "RM8", "RM9"], c: 2 },

                { q: "Aina ingin membeli buku berharga RM10. Dia mempunyai sekeping RM5. Berapa banyak lagi wang RM1 yang dia perlukan?", a: ["3 keping", "4 keping", "5 keping", "6 keping"], c: 2 },

                { q: "Dua keping wang kertas RM5 adalah bersamaan dengan...", a: ["RM2", "RM7", "RM10", "RM5"], c: 2 }

            ]

        };


        // GAME STATE

        let currentLevel = 1;

        let currentQuestionIndex = 0;

        let score = 0;

        let unlockedLevels = [1];

        let canAnswer = true;


        // INITIALIZE GAME

        const gameCard = document.getElementById('game-card');

        const progressBar = document.getElementById('progress');

        const toast = document.getElementById('toast');


        function loadQuestion() {

            canAnswer = true;

            const questions = quizData[currentLevel];

            

            // Update progress bar

            const progressPercent = (currentQuestionIndex / questions.length) * 100;

            progressBar.style.width = `${progressPercent}%`;


            if (currentQuestionIndex >= questions.length) {

                showSummary();

                return;

            }


            const currentQuestion = questions[currentQuestionIndex];

            

            let optionsHTML = '';

            currentQuestion.a.forEach((option, index) => {

                optionsHTML += `

                    <button class="option-btn" onclick="checkAnswer(${index}, this)">

                        ${option}

                    </button>

                `;

            });


            gameCard.innerHTML = `

                <div>

                    <div class="question-meta">Tahap ${currentLevel} • Soalan ${currentQuestionIndex + 1}/${questions.length}</div>

                    <div class="question-text">${currentQuestion.q}</div>

                </div>

                <div class="options-grid">

                    ${optionsHTML}

                </div>

            `;

        }


        function checkAnswer(selectedIndex, buttonElement) {

            if (!canAnswer) return;

            canAnswer = false;


            const questions = quizData[currentLevel];

            const correctAnswerIndex = questions[currentQuestionIndex].c;

            const allButtons = document.querySelectorAll('.option-btn');


            if (selectedIndex === correctAnswerIndex) {

                // Betul

                buttonElement.classList.add('correct');

                score++;

                

                // Efek Visual Tahniah

                toast.classList.add('show');

                setTimeout(() => {

                    toast.classList.remove('show');

                }, 800);

            } else {

                // Salah

                buttonElement.classList.add('wrong');

                allButtons[correctAnswerIndex].classList.add('correct'); // Tunjuk jawapan betul

            }


            // Pandang ke soalan seterusnya selepas kelewatan singkat

            setTimeout(() => {

                currentQuestionIndex++;

                loadQuestion();

            }, 1500);

        }


        function showSummary() {

            progressBar.style.width = '100%';

            const passed = score >= 4; // Syarat wajib: 4/5 markah

            

            let summaryHTML = '';


            if (passed) {

                summaryHTML = `

                    <div class="summary-card">

                        <div class="summary-icon">🏆</div>

                        <div class="summary-title pass">TAHNIAH! LULUS</div>

                        <div class="score-badge">Anda mendapat <strong>${score}/5</strong> markah.</div>

                        <p style="color: var(--text-muted); margin-bottom: 25px;">

                            ${currentLevel < 3 ? 'Tahap seterusnya telah dibuka! Bersedia untuk cabaran baru?' : 'Hebat! Anda telah menamatkan semua tahap dengan cemerlang!'}

                        </p>

                        <button class="action-btn" onclick="nextAction(true)">

                            ${currentLevel < 3 ? 'TERUSKAN KE TAHAP SETERUSNYA' : 'MAIN SEMULA KUIZ'}

                        </button>

                    </div>

                `;

                // Unlock tahap seterusnya secara dinamik

                if (currentLevel < 3 && !unlockedLevels.includes(currentLevel + 1)) {

                    unlockedLevels.push(currentLevel + 1);

                    updateLevelHubUI();

                }

            } else {

                summaryHTML = `

                    <div class="summary-card">

                        <div class="summary-icon">🔄</div>

                        <div class="summary-title fail">CUBA LAGI</div>

                        <div class="score-badge">Anda mendapat <strong>${score}/5</strong> markah.</div>

                        <p style="color: var(--text-muted); margin-bottom: 25px;">

                            Anda perlukan sekurang-kurangnya 4/5 markah untuk unlock tahap seterusnya. Jangan putus asa!

                        </p>

                        <button class="action-btn" onclick="nextAction(false)">

                            CUBA SEMULA TAHAP INI

                        </button>

                    </div>

                `;

            }


            gameCard.innerHTML = summaryHTML;

        }


        function nextAction(isPass) {

            if (isPass) {

                if (currentLevel < 3) {

                    currentLevel++;

                } else {

                    // Reset keseluruhan game jika dah tamat tahap 3

                    currentLevel = 1;

                    unlockedLevels = [1];

                    updateLevelHubUI();

                }

            }

            // Reset trackers untuk tahap semasa/baru

            currentQuestionIndex = 0;

            score = 0;

            updateLevelHubUI();

            loadQuestion();

        }


        function switchLevel(levelNum) {

            if (!unlockedLevels.includes(levelNum)) return; // Blok jika masih lock

            

            currentLevel = levelNum;

            currentQuestionIndex = 0;

            score = 0;

            updateLevelHubUI();

            loadQuestion();

        }


        function updateLevelHubUI() {

            for (let i = 1; i <= 3; i++) {

                const btn = document.getElementById(`btn-lvl${i}`);

                

                // Reset kelas penanda

                btn.className = 'level-btn';

                

                if (unlockedLevels.includes(i)) {

                    btn.classList.add('unlocked');

                } else {

                    btn.classList.add('locked');

                }


                if (currentLevel === i) {

                    btn.classList.add('active');

                }

            }

        }


        // Jalankan game buat kali pertama

        updateLevelHubUI();

        loadQuestion();

    </script>
